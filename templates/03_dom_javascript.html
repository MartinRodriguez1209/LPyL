🌐 Template — DOM con JavaScript

/* El DOM (Document Object Model) es la representación del HTML como objetos de JS */

1. Seleccionar elementos

// ── Por ID ──────────────────────────────────────────────────
const el = document.getElementById('miId');

// ── Por selector CSS (devuelve el PRIMERO) ───────────────────
const el = document.querySelector('.clase');
const el = document.querySelector('#miId');
const el = document.querySelector('div > p');

// ── Por selector CSS (devuelve TODOS como NodeList) ──────────
const lista = document.querySelectorAll('li');
lista.forEach(item => console.log(item));

// ── Otros selectores (menos usados) ─────────────────────────
document.getElementsByClassName('clase');  // HTMLCollection
document.getElementsByTagName('p');        // HTMLCollection

// ── Buscar dentro de un elemento ────────────────────────────
const contenedor = document.querySelector('.contenedor');
const hijo = contenedor.querySelector('p'); // busca dentro

2. Leer y modificar contenido

Texto original del párrafo

// ── textContent → solo texto plano (más seguro) ──────────────
const p = document.getElementById('miParrafo');
p.textContent = 'Nuevo texto';
console.log(p.textContent);

// ── innerHTML → texto + etiquetas HTML ───────────────────────
p.innerHTML = '<strong>Texto en negrita</strong>';

// ── value → para inputs, selects y textareas ─────────────────
const input = document.querySelector('input');
input.value = 'nuevo valor';
console.log(input.value);

3. Modificar estilos y clases

Este elemento cambia de estilo

const caja = document.getElementById('caja');

// ── Estilo inline ────────────────────────────────────────────
caja.style.color = 'red';
caja.style.backgroundColor = '#333';   // camelCase
caja.style.fontSize = '1.5rem';
caja.style.display = 'none';           // ocultar

// ── Clases (lo más recomendado) ──────────────────────────────
caja.classList.add('activo');          // agrega clase
caja.classList.remove('activo');       // quita clase
caja.classList.toggle('activo');       // alterna (add/remove)
caja.classList.contains('activo');     // true/false
caja.classList.replace('old', 'new'); // reemplaza

4. Modificar atributos

const img  = document.querySelector('img');
const link = document.querySelector('a');

// ── getAttribute / setAttribute / removeAttribute ────────────
img.getAttribute('src');                     // leer
img.setAttribute('src', 'nueva.jpg');       // escribir
img.removeAttribute('alt');                 // eliminar
img.hasAttribute('disabled');              // true/false

// ── Acceso directo para atributos comunes ────────────────────
link.href = 'https://google.com';
img.src    = 'foto.jpg';
img.alt    = 'descripción';
input.disabled = true;
input.checked  = false;
input.value    = 'texto';

// ── data-* atributos ─────────────────────────────────────────
// HTML: <div data-id="42" data-nombre="Carlos">
const div = document.querySelector('div');
div.dataset.id;        // "42"
div.dataset.nombre;    // "Carlos"
div.dataset.nuevoVal = 'hola'; // agrega data-nuevo-val
